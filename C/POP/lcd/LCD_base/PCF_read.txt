// Ler a porta do PCF
int PCF_read(void){
    int dado;
    UCB0CTL1 &= ~UCTR;     //Mestre RX
    UCB0CTL1 |= UCTXSTT;   //Gerar START
    while ( (UCB0CTL1 & UCTXSTT) == UCTXSTT);
    UCB0CTL1 |= UCTXSTP;    //Gerar STOP + NACK
    while ( (UCB0IFG & UCRXIFG) == 0)   ;          //Esperar RX
    dado=UCB0RXBUF;
    delay(50);
    return dado;
}
